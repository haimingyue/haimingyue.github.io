微信授权跳转

Node 端跳转到微信端，获取 openId。

node 端获取到 openId 之后，向客户端写入 Cookie。

后端代码

一、第一步

router.get("/redirect", function(req, res) {
  let redirectUrl = req.query.url;
  cache.put("redirectUrl", redirectUrl);
  let scope = req.query.scope;
  let callback = "http%3A%2F%2Fm.imooc.com%2Fapi%2Fwechat%2FgetOpenId";
  let authorizeUrl = `https://open.weixin.qq.com/connect/oauth2/authorize?appid=${config.appId}&redirect_uri=${callback}&response_type=code&scope=${scope}&state=STATE#wechat_redirect`;
  res.redirect(authorizeUrl);
});

解释：

  • 前端访问/api/wechat/redirect接口，带着 url 的参数和 scope 的参数
  • 获取这两个参数，并且把 url 保存（cache）起来
  • 声明回调地址–> 第二个后端接口getOpenID
  • 重定向到微信的地址authorizeUrl,以获取 code。

二、第二步：根据 code 码获取 appId，种植到前端 cookie

router.get("/getOpenId", function(req, res) {
  let code = req.query.code;
  let token_url = `https://api.weixin.qq.com/sns/oauth2/access_token?appid=${config.appId}&secret=${config.appsecret}&code=${code}&grant_type=authorization_code`;
  if (!code) {
    res.json({
      code: "1001",
      data: "",
      message: "未获取到code授权码",
    });
  } else {
    request.get(token_url, function(error, response, body) {
      if (!error && response.statusCode === 200) {
        let data = JSON.parse(body);
        let expire_time = 1000 * 60 * 1;
        res.cookie("openId", data.openid, { maxAge: expire_time });
        let redirectUrl = cache.get("redirectUrl");
        console.log("redirectUrl", redirectUrl);
        res.redirect(redirectUrl);
      } else {
        // console.log('err', error)
      }
    });
  }
});

解释：

  • 根据获取的 code 码，去再次访问微信提供的接口token_url,参数参见文档2 第二步：通过 code 换取网页授权 access_token

  • 获取到 OpenId 之后，把 cookie 种植到前端

    res.cookie('openId', data.openid, { maxAge: expire_time })

  • 然后重定向到前端页面res.redirect(redirectUrl)

重新整理，封装公用方法

util 封装微信接口状态码处理

module.exports = {
  // 对微信接口的状态码等统一处理
  handleResponse(error, response, body) {
    if (!error && response.statusCode === 200) {
      // 对请求返回统一处理
      let data = JSON.parse(body);
      console.log("body::", data);
      if (data && !data.errcode) {
        return this.handleSuc(data);
      } else {
        return this.handleFail(data.errmessage, data.errcode);
      }
    } else {
      return this.handleFail(error, 10009);
    }
  },
  // 返回成功的数据封装
  handleSuc(data = "") {
    return {
      code: 0,
      data,
      message: "",
    };
  },
  // 返回失败的数据封装
  handleFail(message = "", code = 10001) {
    return {
      code,
      data: "",
      message,
    };
  },
};

// common.js
exports.getUserInfo = function(access_token, openId) {
  return new Promise((resolve, reject) => {
    let userinfo = `https://api.weixin.qq.com/sns/userinfo?access_token=${access_token}&openid=${openId}&lang=zh_CN`;
    request.get(userinfo, (err, response, bocy) => {
      let result = util.handleResponse(err, response, bocy);
      resolve(result);
    });
  });
};
// 路由定义
router.get("/getUserInfo", async (req, res) => {

获取基础 token 官方文档

// 获取基础token

exports.getToken = function() {
  return new Promise((resolve, reject) => {
    let token = `https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid=${config.appId}&secret=${config.appsecret}`;
    request.get(token, (err, response, bocy) => {
      let result = util.handleResponse(err, response, bocy);
      resolve(result);
    });
  });
};

2、获取 ticket

// 获取ticket
exports.getTicket = function(token) {
  return new Promise((resolve, reject) => {
    let tokenUrl = `https://api.weixin.qq.com/cgi-bin/ticket/getticket?access_token=${token}&type=jsapi`;
    request.get(tokenUrl, (err, response, bocy) => {
      let result = util.handleResponse(err, response, bocy);
      resolve(result);
    });
  });
};

// 生成随机数
createNonceStr() {
  return Math.random().toString(36).substr(2, 15)

    return parseInt(new Date().getTime() / 1000) + ''

4、对参数排序

// object 转换成json并且排序
raw(args) {
  let keys = Object.keys(args).sort();
  let obj = {};
  keys.forEach(key => {
    obj[key] = args[key]
  })
  // 将对象转成key value形式
  let val = '';
  for (let key in obj) {
    val += '&' + key + '=' + obj[key]
  }
  return val.substr(1)
},

router.get("/jssdk", async (req, res) => {
  console.log("res", req.query);
  let url = req.query.url;
  let result = await common.getToken();
  if (result.code == 0) {
    let token = result.data.access_token;
    let result2 = await common.getTicket(token);

    console.log("result2", result2);
    if (result2.code == 0) {
      let data = result2.data;
      let params = {
        noncestr: util.createNonceStr,
        jsapi_ticket: data.ticket,
        timestamp: util.createTimeStamp,
        url,
      };
      let str = util.raw(params);
      let sign = createHash("sha1")
        .update(str)
        .digest("hex");
      res.json(
        util.handleSuc({
          appId: config.appId, // 必填，公众号的唯一标识
          timestamp: params.timestamp, // 必填，生成签名的时间戳
          noncestr: params.noncestr, // 必填，生成签名的随机串
          signature: sign, // 必填，签名
          jsApiList: [
            "updateAppMessageShareData",
            "updateTimelineShareData",
            "chooseWXPay",
          ], // 必填，需要使用的JS接口列表
        })
      );
    }
  }
});
